Modular javascript with dependancies and inheritance

Child.js

define('Child', function () {
    return Class.extend({
        init: function (id, options) {
            this.el = document.getElementById(id);
            this.options = options;
        },
        get: function () {
            return this.options;
        },
        childMethod: function () {
            return true;
        }
    });
});

Parent.js

define('Parent', ['Child'], function (Child) {
    return Child.extend({
        init: function (id, options) {
            this.el = document.getElementById(id);
            this.options = options;
            this._super(id, options);
        },
        get: function () {
            // Call the inherited version of get()
            return this._super();
        },
        parentMethod: function () {
            return true;
        }
    });
});
